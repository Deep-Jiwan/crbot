Docker Compose Guide
--------------------

# Start compose with build from local
docker-compose -p crbot-build -f docker-compose-build.yaml up -d --build

# Start compose using images from GitHub
docker-compose -p crbot -f docker-compose.yaml up -d

# Stop existing compose (data not deleted)
docker-compose -p crbot -f docker-compose.yaml stop
docker-compose -p crbot-build -f docker-compose-build.yaml stop

# Start existing compose (no rebuild)
docker-compose -p crbot -f docker-compose.yaml up -d
docker-compose -p crbot-build -f docker-compose-build.yaml up -d

# Remove compose and delete data
docker-compose -p crbot -f docker-compose.yaml down --volumes
docker-compose -p crbot-build -f docker-compose-build.yaml down --volumes
